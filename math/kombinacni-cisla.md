    import itertools

    def comb(n, k):
        return len(list(itertools.combinations(range(n), k)))

    res = []
    for n in xrange(11):
        res2 = []
        for k in xrange(11):
            if n >= k:
                res2.append((n, k, comb(n, k)))
            else:
                res2.append(None)
        res.append(res2)

    print '<table>',
    for row in res:
        print '<tr><td>',
        print '</td><td>'.join(map(lambda x: '${%s \\choose %s} = %s$'% x if x else '', row)),
        print '</td></tr>',
    print '</table>'
